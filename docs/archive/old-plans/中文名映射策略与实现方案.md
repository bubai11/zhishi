# 植物科普系统：学名到中文名映射策略与实现方案

## 文档说明

本文档包含完整的学名到中文名映射策略设计，以及可供 Claude/Codex 执行的详细技术方案。

---

## 一、任务概述

### 1.1 背景

系统已完成数据完整性检查，plants 表已有 `scientific_name` 字段，但 `chinese_name` 字段为空。需要建立映射策略，将英文学名转换为中文名称。

### 1.2 目标

| 目标 | 说明 |
|------|------|
| **主目标** | 为 plants 表中所有物种填充中文名称 |
| **次目标** | 建立可复用的中文名映射服务，支持后续增量更新 |
| **质量要求** | 核心植物 100% 准确，其他植物 95% 以上准确率 |

---

## 二、映射策略设计

### 2.1 三层映射架构

```text
┌─────────────────────────────────────────────────────────────────┐
│ Layer 1: 精确匹配层（优先级最高） │
├─────────────────────────────────────────────────────────────────┤
│ 数据源： │
│ 1. 核心植物字典（手动整理）→ 100% 准确 │
│ 2. 中国植物志名录缓存 → 99% 准确 │
│ 3. 植物智 API 实时查询 → 95% 准确 │
└─────────────────────────────────────────────────────────────────┘
↓ 未匹配
┌─────────────────────────────────────────────────────────────────┐
│ Layer 2: 模糊匹配层 │
├─────────────────────────────────────────────────────────────────┤
│ 规则引擎： │
│ 1. 属名映射：Rosa → 蔷薇属 │
│ 2. 种加词翻译：chinensis → 中国 │
│ 3. 组合规则：Rosa chinensis → 月季花（特殊处理） │
└─────────────────────────────────────────────────────────────────┘
↓ 未匹配
┌─────────────────────────────────────────────────────────────────┐
│ Layer 3: 降级处理层 │
├─────────────────────────────────────────────────────────────────┤
│ 1. 保留英文学名作为显示 │
│ 2. 标记待补充，后台人工补录 │
└─────────────────────────────────────────────────────────────────┘
```

### 2.2 数据源优先级

| 优先级 | 数据源 | 覆盖范围 | 准确度 | 获取方式 |
|--------|--------|----------|--------|----------|
| **P0** | 核心植物字典 | 50-100种 | 100% | 手动整理 |
| **P1** | 植物智 API | 全 | 95% | 按需查询 |
| **P2** | 属名映射规则 | 全属 | 80% | 规则引擎 |
| **P3** | 保留英文 | 100% | N/A | 降级处理 |

### 3.2 创建中文字典缓存表

```sql
-- 用于缓存查询结果，加速后续访问
CREATE TABLE IF NOT EXISTS `chinese_name_cache` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `scientific_name` VARCHAR(200) NOT NULL COMMENT '学名',
    `chinese_name` VARCHAR(100) NOT NULL COMMENT '中文名',
    `source` VARCHAR(20) NOT NULL COMMENT '来源: manual/api/rule',
    `confidence` TINYINT DEFAULT 80 COMMENT '置信度',
    `hit_count` INT DEFAULT 0 COMMENT '命中次数',
    `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_scientific` (`scientific_name`),
    INDEX `idx_chinese` (`chinese_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
```

## 五、核心服务实现

### 5.1 中文名映射服务

**文件路径:** `services/ChineseNameService.js`

```javascript
// services/ChineseNameService.js
const db = require('../config/database');
const axios = require('axios');
const coreDict = require('../data/core_plants_dict');
const genusMapping = require('../data/genus_mapping');

class ChineseNameService {
    
    /**
     * 主入口：获取学名的中文名
     * @param {string} scientificName - 学名
     * @returns {Promise<Object>} { chineseName, source, confidence }
     */
    async getChineseName(scientificName) {
        // Layer 1: 检查缓存
        const cached = await this.checkCache(scientificName);
        if (cached) return cached;
        
        // Layer 2: 核心字典（手动整理）
        const core = this.checkCoreDict(scientificName);
        if (core) {
            await this.saveCache(scientificName, core.chineseName, 'manual', 100);
            return core;
        }
        
        // Layer 3: 植物智 API
        const apiResult = await this.queryIPlant(scientificName);
        if (apiResult) {
            await this.saveCache(scientificName, apiResult.chineseName, 'api', 95);
            return apiResult;
        }
        
        // Layer 4: 规则匹配
        const ruleResult = this.applyRules(scientificName);
        if (ruleResult && ruleResult.chineseName) {
            await this.saveCache(scientificName, ruleResult.chineseName, 'rule', ruleResult.confidence);
            return ruleResult;
        }
        
        // Layer 5: 降级处理
        return {
            chineseName: null,
            source: 'none',
            confidence: 0
        };
    }
    
    /**
     * 检查缓存
     */
    async checkCache(scientificName) {
        const [rows] = await db.execute(
            'SELECT chinese_name, source, confidence FROM chinese_name_cache WHERE scientific_name = ?',
            [scientificName]
        );
        if (rows.length > 0) {
            // 更新命中次数
            await db.execute(
                'UPDATE chinese_name_cache SET hit_count = hit_count + 1 WHERE scientific_name = ?',
                [scientificName]
            );
            return {
                chineseName: rows[0].chinese_name,
                source: rows[0].source,
                confidence: rows[0].confidence
            };
        }
        return null;
    }
    
    /**
     * 检查核心字典
     */
    checkCoreDict(scientificName) {
        const name = scientificName.trim();
        
        // 精确匹配
        if (coreDict[name]) {
            return { chineseName: coreDict[name], source: 'manual', confidence: 100 };
        }
        
        // 匹配属名（用于属级显示）
        const genus = name.split(' ')[0];
        if (coreDict[genus]) {
            return { chineseName: coreDict[genus], source: 'manual', confidence: 100 };
        }
        
        return null;
    }
    
    /**
     * 查询植物智 API
     */
    async queryIPlant(scientificName) {
        try {
            const url = `http://www.iplant.cn/api/sp/sci/${encodeURIComponent(scientificName)}`;
            const response = await axios.get(url, { 
                timeout: 8000,
                headers: { 'User-Agent': 'Mozilla/5.0' }
            });
            
            if (response.data && response.data.code === 200 && response.data.result) {
                const cnName = response.data.result.chinese_name || 
                              response.data.result.cnName || 
                              response.data.result.cname;
                if (cnName && cnName !== scientificName) {
                    return { chineseName: cnName, source: 'api', confidence: 95 };
                }
            }
            return null;
        } catch (error) {
            console.error(`[IPlant] ${scientificName} 查询失败:`, error.message);
            return null;
        }
    }
    
    /**
     * 应用规则匹配
     */
    applyRules(scientificName) {
        const parts = scientificName.trim().split(' ');
        const genus = parts[0];
        const epithet = parts[1];
        
        // 检查特殊组合
        const fullName = `${genus} ${epithet}`;
        if (genusMapping[fullName]) {
            return { 
                chineseName: genusMapping[fullName], 
                source: 'rule', 
                confidence: 90 
            };
        }
        
        // 检查属名映射
        const genusCn = genusMapping[genus];
        if (!genusCn) return null;
        
        // 如果有种加词，尝试翻译
        if (epithet && genusMapping[epithet]) {
            const epithetCn = genusMapping[epithet];
            return { 
                chineseName: `${genusCn} (${epithetCn})`, 
                source: 'rule', 
                confidence: 70 
            };
        }
        
        // 只有属名
        return { chineseName: genusCn, source: 'rule', confidence: 80 };
    }
    
    /**
     * 保存到缓存
     */
    async saveCache(scientificName, chineseName, source, confidence) {
        await db.execute(`
            INSERT INTO chinese_name_cache (scientific_name, chinese_name, source, confidence) 
            VALUES (?, ?, ?, ?)
            ON DUPLICATE KEY UPDATE
            chinese_name = VALUES(chinese_name),
            source = VALUES(source),
            confidence = VALUES(confidence),
            updated_at = NOW()
        `, [scientificName, chineseName, source, confidence]);
    }
    
    /**
     * 批量更新 plants 表的 chinese_name
     * @param {number} limit - 每批处理数量
     * @param {number} offset - 起始偏移
     */
    async batchUpdatePlants(limit = 100, offset = 0) {
        const [plants] = await db.execute(`
            SELECT id, scientific_name 
            FROM plants 
            WHERE (chinese_name IS NULL OR chinese_name = '')
              AND scientific_name IS NOT NULL
            LIMIT ? OFFSET ?
        `, [limit, offset]);
        
        let updated = 0;
        let failed = [];
        
        for (let i = 0; i < plants.length; i++) {
            const plant = plants[i];
            try {
                const result = await this.getChineseName(plant.scientific_name);
                if (result.chineseName) {
                    await db.execute(`
                        UPDATE plants 
                        SET chinese_name = ?, 
                            translation_source = ?, 
                            translation_confidence = ?
                        WHERE id = ?
                    `, [result.chineseName, result.source, result.confidence, plant.id]);
                    updated++;
                    console.log(`[${i+1}/${plants.length}] ${plant.scientific_name} → ${result.chineseName}`);
                } else {
                    failed.push(plant.scientific_name);
                    console.log(`[${i+1}/${plants.length}] ${plant.scientific_name} → 未找到`);
                }
            } catch (err) {
                failed.push(plant.scientific_name);
                console.error(`[${i+1}/${plants.length}] ${plant.scientific_name} 处理失败:`, err.message);
            }
            
            // 避免请求过快
            await new Promise(resolve => setTimeout(resolve, 300));
        }
        
        return { updated, total: plants.length, failed };
    }
    
    /**
     * 获取未匹配植物列表
     */
    async getUnmatchedPlants(limit = 100) {
        const [rows] = await db.execute(`
            SELECT id, scientific_name 
            FROM plants 
            WHERE (chinese_name IS NULL OR chinese_name = '')
              AND scientific_name IS NOT NULL
            LIMIT ?
        `, [limit]);
        return rows;
    }
    
    /**
     * 获取统计信息
     */
    async getStatistics() {
        const [total] = await db.execute('SELECT COUNT(*) as count FROM plants WHERE scientific_name IS NOT NULL');
        const [matched] = await db.execute('SELECT COUNT(*) as count FROM plants WHERE chinese_name IS NOT NULL');
        const [bySource] = await db.execute(`
            SELECT translation_source, COUNT(*) as count 
            FROM plants 
            WHERE chinese_name IS NOT NULL 
            GROUP BY translation_source
        `);
        
        return {
            total: total[0].count,
            matched: matched[0].count,
            unmatched: total[0].count - matched[0].count,
            bySource
        };
    }
}

module.exports = new ChineseNameService();
```

### 5.2 批量更新脚本

**文件路径:** `scripts/batch-update-chinese-names.js`

```javascript
// scripts/batch-update-chinese-names.js
/**
 * 批量更新植物中文名脚本
 * 用法: node scripts/batch-update-chinese-names.js [--batch=100] [--resume]
 */

const db = require('../config/database');
const ChineseNameService = require('../services/ChineseNameService');
const fs = require('fs');
const path = require('path');

// 解析命令行参数
const args = process.argv.slice(2);
let batchSize = 100;
let resume = false;

args.forEach(arg => {
    if (arg.startsWith('--batch=')) {
        batchSize = parseInt(arg.split('=')[1]);
    }
    if (arg === '--resume') {
        resume = true;
    }
});

// 日志文件路径
const LOG_DIR = path.join(__dirname, '../logs');
const ERROR_LOG = path.join(LOG_DIR, 'translation-errors.log');

// 确保日志目录存在
if (!fs.existsSync(LOG_DIR)) {
    fs.mkdirSync(LOG_DIR, { recursive: true });
}

// 记录错误日志
function logError(scientificName, error) {
    const logLine = `[${new Date().toISOString()}] ${scientificName}: ${error}\n`;
    fs.appendFileSync(ERROR_LOG, logLine);
}

// 主函数
async function main() {
    console.log('=== 植物中文名批量更新脚本 ===');
    console.log(`批次大小: ${batchSize}`);
    console.log(`断点续传: ${resume ? '是' : '否'}`);
    console.log('开始时间:', new Date().toLocaleString());
    console.log('---');
    
    let offset = 0;
    let totalUpdated = 0;
    let hasMore = true;
    
    while (hasMore) {
        console.log(`\n处理第 ${offset / batchSize + 1} 批...`);
        
        const result = await ChineseNameService.batchUpdatePlants(batchSize, offset);
        
        totalUpdated += result.updated;
        console.log(`本批: 处理 ${result.total} 条, 更新 ${result.updated} 条`);
        
        // 记录失败的学名
        if (result.failed && result.failed.length > 0) {
            result.failed.forEach(name => {
                logError(name, '未找到中文名或API调用失败');
            });
        }
        
        if (result.total < batchSize) {
            hasMore = false;
        } else {
            offset += batchSize;
        }
        
        // 短暂停顿
        await new Promise(resolve => setTimeout(resolve, 1000));
    }
    
    console.log('\n---');
    console.log('完成时间:', new Date().toLocaleString());
    console.log(`总计更新: ${totalUpdated} 条`);
    
    // 输出统计
    const stats = await ChineseNameService.getStatistics();
    console.log('\n--- 统计信息 ---');
    console.log(`总植物数: ${stats.total}`);
    console.log(`已匹配中文: ${stats.matched}`);
    console.log(`未匹配: ${stats.unmatched}`);
    console.log(`\n按来源分布:`);
    stats.bySource.forEach(s => {
        console.log(`  ${s.translation_source}: ${s.count} 条`);
    });
    
    process.exit(0);
}

// 运行主函数
main().catch(err => {
    console.error('脚本执行失败:', err);
    process.exit(1);
});
```

```javascript
        const [result] = await db.execute(`
            UPDATE plants 
            SET chinese_name = ?, translation_source = ?, translation_confidence = 100
            WHERE id = ?
        `, [chineseName, source, plantId]);
        
        if (result.affectedRows === 0) {
            return res.status(404).json({ code: 404, message: '植物不存在' });
        }
        
        // 同时更新缓存
        const [plant] = await db.execute('SELECT scientific_name FROM plants WHERE id = ?', [plantId]);
        if (plant[0]?.scientific_name) {
            await db.execute(`
                INSERT INTO chinese_name_cache (scientific_name, chinese_name, source, confidence)
                VALUES (?, ?, ?, 100)
                ON DUPLICATE KEY UPDATE
                chinese_name = VALUES(chinese_name),
                source = VALUES(source)
            `, [plant[0].scientific_name, chineseName, source]);
        }
        
        res.json({ code: 200, message: '更新成功' });
    } catch (error) {
        console.error('手动更新失败:', error);
        res.status(500).json({ code: 500, message: '服务器错误' });
    }
});

module.exports = router;
```
