代码片段（预警等级判断）：

```js
function determineAlertLevel(prevCategory, currCategory, changeType) {
  if (currCategory === 'CR' && changeType === 'upgraded') return 'high';
  if (['EN', 'CR'].includes(currCategory)) return 'high';
  if (currCategory === 'VU') return 'medium';
  return 'low';
}
```

```js
function decodeItems(items) {
  return items
    .filter(item => item.hasdesc && item.desc)
    .map(item => {
      const text = Buffer.from(item.desc, 'base64').toString('utf8').trim();
      return `【${item.t}】${text}`;
    })
    .join('\n');
}
```
