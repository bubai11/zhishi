# 植物科普系统：数据补全方案（cover_image / category / plant_ecology）

## 文档说明

本文档提供三个核心数据表的补全方案：`plants.cover_image`、`plants.category`、`plant_ecology`。当前填充率均为 0%，需系统性补全以满足前端联调需求。

---

## 一、数据补全策略总览

| 字段 | 当前状态 | 目标 | 补全策略 | 优先级 |
|------|---------|------|---------|--------|
| `cover_image` | 0% | ≥80% | 植物智API + Unsplash回退 + 占位符 | 🔴 高 |
| `category` | 0% | 100% | 基于科属规则分类 + 手工标注核心种 | 🔴 高 |
| `plant_ecology` | 空表 | ≥50种 | 规则生成 + 文献查询 | 🟡 中 |

## 二、cover_image 图片数据补全

### 2.1 数据来源优先级

| 优先级 | 来源 | 说明 | 成功率预估 |
|--------|------|------|-----------|
| 1 | 植物智API | 官方平台，图片质量高 | 约30-40% |
| 2 | Unsplash | 免费图库，按关键词搜索 | 约60-70% |
| 3 | 本地占位符 | 按科属生成彩色占位图 | 100% |

### 2.2 植物智API调用封装

```javascript
// services/imageService.js
const axios = require('axios');
const db = require('../config/database');

class ImageService {
    
    /**
     * 从植物智API获取植物图片
     * @param {string} scientificName - 学名
     * @returns {Promise<Array>} 图片URL数组
     */
    async fetchFromIPlant(scientificName) {
        try {
            const url = `http://www.iplant.cn/api/sp/sci/${encodeURIComponent(scientificName)}`;
            const response = await axios.get(url, {
                timeout: 8000,
                headers: { 'User-Agent': 'Mozilla/5.0' }
            });
            
            if (response.data?.code === 200 && response.data?.result?.images) {
                return response.data.result.images;
            }
            return [];
        } catch (error) {
            console.error(`[IPlant] ${scientificName} 图片获取失败:`, error.message);
            return [];
        }
    }
    
    /**
     * 从Unsplash获取回退图片
     * @param {Object} plant - 植物信息
     * @returns {string} 图片URL
     */
    getUnsplashFallback(plant) {
        const keywords = [];
        if (plant.chinese_name) keywords.push(plant.chinese_name);
        if (plant.wcvp_genus) keywords.push(plant.wcvp_genus);
        keywords.push('plant');
        const query = keywords.join(' ');
        return `https://source.unsplash.com/featured/800x600?${encodeURIComponent(query)}`;
    }
    
    /**
     * 获取本地占位图（按科属颜色区分）
     * @param {Object} plant - 植物信息
     * @returns {string} 占位图URL
     */
    getPlaceholder(plant) {
        const colorMap = {
            'Rosaceae': '#e8a87c',
            'Orchidaceae': '#b5e3d5',
            'Pinaceae': '#5d8a6f',
            'Fabaceae': '#c5e0b4',
            'Asteraceae': '#f4d03f',
            'default': '#a8dadc'
        };
        const color = colorMap[plant.wcvp_family] || colorMap.default;
        // 使用图片占位服务
        return `https://via.placeholder.com/800x600/${color.slice(1)}/ffffff?text=${encodeURIComponent(plant.chinese_name || plant.scientific_name)}`;
    }
}

module.exports = new ImageService();

javascript
// scripts/populate-plant-images.js
/**
 * 植物图片批量填充脚本
 * 用法: node scripts/populate-plant-images.js [--limit=100] [--resume]
 */

const db = require('../config/database');
const ImageService = require('../services/imageService');
const fs = require('fs');
const path = require('path');

// 解析命令行参数
const args = process.argv.slice(2);
let limit = 100;
let resume = false;

args.forEach(arg => {
    if (arg.startsWith('--limit=')) {
        limit = parseInt(arg.split('=')[1]);
    }
    if (arg === '--resume') {
        resume = true;
    }
});

// 日志配置
const LOG_DIR = path.join(__dirname, '../logs');
const IMAGE_LOG = path.join(LOG_DIR, 'image-fetch-errors.log');

if (!fs.existsSync(LOG_DIR)) {
    fs.mkdirSync(LOG_DIR, { recursive: true });
}

function logError(scientificName, error) {
    const logLine = `[${new Date().toISOString()}] ${scientificName}: ${error}\n`;
    fs.appendFileSync(IMAGE_LOG, logLine);
}

async function populateImages() {
    console.log('=== 植物图片批量填充脚本 ===');
    console.log(`限制数量: ${limit}`);
    console.log(`断点续传: ${resume ? '是' : '否'}`);
    console.log('开始时间:', new Date().toLocaleString());
    console.log('---');
    
    // 1. 获取需要填充图片的植物
    let whereClause = 'cover_image IS NULL';
    if (resume) {
        whereClause += ' AND cover_image IS NULL';
    }
    
    const [plants] = await db.execute(`
        SELECT id, scientific_name, chinese_name, wcvp_family, wcvp_genus
        FROM plants 
        WHERE ${whereClause}
        ORDER BY chinese_name IS NOT NULL DESC, id
        LIMIT ?
    `, [limit]);
    
    console.log(`待处理植物数: ${plants.length}`);
    
    let success = 0;
    let failed = 0;
    
    for (let i = 0; i < plants.length; i++) {
        const plant = plants[i];
        console.log(`[${i+1}/${plants.length}] 处理: ${plant.scientific_name}`);
        
        let imageUrl = null;
        let source = null;
        
        try {
            // Layer 1: 植物智API
            const images = await ImageService.fetchFromIPlant(plant.scientific_name);
            if (images && images.length > 0) {
                imageUrl = images[0];
                source = 'iplant';
                console.log(`  → 从植物智获取图片`);
            }
            
            // Layer 2: Unsplash回退
            if (!imageUrl) {
                imageUrl = ImageService.getUnsplashFallback(plant);
                source = 'unsplash';
                console.log(`  → 使用Unsplash回退`);
            }
            
            // Layer 3: 更新数据库
            if (imageUrl) {
                await db.execute(
                    'UPDATE plants SET cover_image = ? WHERE id = ?',
                    [imageUrl, plant.id]
                );
                
                // 同时插入 media_assets 记录
                const [assetResult] = await db.execute(`
                    INSERT INTO media_assets (kind, url, metadata)
                    VALUES ('image', ?, ?)
                `, [imageUrl, JSON.stringify({ source, plant_id: plant.id, scientific_name: plant.scientific_name })]);
                
                await db.execute(`
                    INSERT INTO plant_media (plant_id, media_asset_id, sort_order)
                    VALUES (?, ?, 0)
                `, [plant.id, assetResult.insertId]);
                
                success++;
            } else {
                // 使用本地占位符
                const placeholder = ImageService.getPlaceholder(plant);
                await db.execute(
                    'UPDATE plants SET cover_image = ? WHERE id = ?',
                    [placeholder, plant.id]
                );
                console.log(`  → 使用占位图`);
                success++;
            }
            
        } catch (error) {
            failed++;
            logError(plant.scientific_name, error.message);
            console.error(`  ❌ 失败: ${error.message}`);
        }
        
        // 避免请求过快
        await new Promise(resolve => setTimeout(resolve, 300));
    }
    
    console.log('\n---');
    console.log(`✅ 成功: ${success}`);
    console.log(`❌ 失败: ${failed}`);
    console.log(`完成时间: ${new Date().toLocaleString()}`);
    
    // 输出统计
    const [stats] = await db.execute(`
        SELECT COUNT(*) as total, COUNT(cover_image) as with_image 
        FROM plants
    `);
    console.log(`\n当前覆盖率: ${stats[0].with_image}/${stats[0].total} (${(stats[0].with_image/stats[0].total*100).toFixed(2)}%)`);
}

populateImages().catch(console.error);

// scripts/populate-plant-category.js
/**
 * 植物分类标签填充脚本
 * 用法: node scripts/populate-plant-category.js
 */

const db = require('../config/database');
const categoryRules = require('../config/categoryRules');

async function populateCategory() {
    console.log('=== 植物分类标签填充脚本 ===');
    console.log('开始时间:', new Date().toLocaleString());
    console.log('---');
    
    // 1. 获取所有植物
    const [plants] = await db.execute(`
        SELECT id, scientific_name, chinese_name, wcvp_family, wcvp_genus
        FROM plants
        WHERE category IS NULL
    `);
    
    console.log(`待处理植物数: ${plants.length}`);
    
    let stats = {
        Tropical: 0,
        Temperate: 0,
        Arid: 0,
        Aquatic: 0,
        Medicinal: 0,
        Ornamental: 0,
        Uncategorized: 0
    };
    
    for (const plant of plants) {
        let category = 'Uncategorized';
        
        // 规则1: 热带植物
        if (categoryRules.tropical.families.includes(plant.wcvp_family) ||
            categoryRules.tropical.genera.includes(plant.wcvp_genus)) {
            category = 'Tropical';
        }
        // 规则2: 温带植物
        else if (categoryRules.temperate.families.includes(plant.wcvp_family) ||
                 categoryRules.temperate.genera.includes(plant.wcvp_genus)) {
            category = 'Temperate';
        }
        // 规则3: 旱生植物
        else if (categoryRules.arid.families.includes(plant.wcvp_family) ||
                 categoryRules.arid.genera.includes(plant.wcvp_genus)) {
            category = 'Arid';
        }
        // 规则4: 水生植物
        else if (categoryRules.aquatic.families.includes(plant.wcvp_family) ||
                 categoryRules.aquatic.genera.includes(plant.wcvp_genus)) {
            category = 'Aquatic';
        }
        
        // 更新数据库
        await db.execute(
            'UPDATE plants SET category = ? WHERE id = ?',
            [category, plant.id]
        );
        
        stats[category]++;
        
        if (stats[category] % 50 === 0) {
            console.log(`  已处理 ${stats.Tropical + stats.Temperate + stats.Arid + stats.Aquatic + stats.Uncategorized} 条`);
        }
    }
    
    console.log('\n---');
    console.log('分类统计:');
    console.log(`  🌴 Tropical: ${stats.Tropical}`);
    console.log(`  🍂 Temperate: ${stats.Temperate}`);
    console.log(`  🏜️ Arid: ${stats.Arid}`);
    console.log(`  💧 Aquatic: ${stats.Aquatic}`);
    console.log(`  🌿 Uncategorized: ${stats.Uncategorized}`);
    console.log(`完成时间: ${new Date().toLocaleString()}`);
}

populateCategory().catch(console.error);

4.3 生态数据填充脚本
javascript
// scripts/populate-plant-ecology.js
/**
 * 植物生态数据填充脚本
 * 用法: node scripts/populate-plant-ecology.js
 */

const db = require('../config/database');
const coreEcology = require('../data/coreEcology');

// 科属调整规则
const adjustmentRules = {
    tropical: {
        families: ['Orchidaceae', 'Rubiaceae', 'Arecaceae', 'Musaceae'],
        adjustments: { temperature_tolerance: 30, cold_tolerance: -20, light_tolerance: 20, water_requirement: 20 }
    },
    temperate: {
        families: ['Rosaceae', 'Pinaceae', 'Salicaceae', 'Betulaceae'],
        adjustments: { cold_tolerance: 20, temperature_tolerance: -10 }
    },
    arid: {
        families: ['Cactaceae', 'Agavaceae', 'Asphodelaceae'],
        adjustments: { drought_tolerance: 40, water_requirement: -30, light_tolerance: 20 }
    },
    aquatic: {
        families: ['Nymphaeaceae', 'Hydrocharitaceae'],
        adjustments: { water_requirement: 40, drought_tolerance: -30 }
    }
};

function clamp(value, min = 0, max = 100) {
    return Math.max(min, Math.min(max, value));
}

function generateEcology(plant) {
    // 1. 检查核心植物手工标注
    if (coreEcology[plant.scientific_name]) {
        return coreEcology[plant.scientific_name];
    }
    
    // 2. 基于科属生成默认值
    let ecology = {
        light_tolerance: 60,
        water_requirement: 50,
        temperature_tolerance: 50,
        drought_tolerance: 50,
        cold_tolerance: 50
    };
    
    // 3. 应用调整规则
    for (const [type, rule] of Object.entries(adjustmentRules)) {
        if (rule.families.includes(plant.wcvp_family)) {
            for (const [key, delta] of Object.entries(rule.adjustments)) {
                ecology[key] = clamp(ecology[key] + delta);
            }
        }
    }
    
    return ecology;
}

async function populateEcology() {
    console.log('=== 植物生态数据填充脚本 ===');
    console.log('开始时间:', new Date().toLocaleString());
    console.log('---');
    
    // 1. 获取需要填充生态数据的植物
    const [plants] = await db.execute(`
        SELECT id, scientific_name, chinese_name, wcvp_family
        FROM plants
        WHERE EXISTS (SELECT 1 FROM plants LIMIT 1)
        LIMIT 500
    `);
    
    console.log(`待处理植物数: ${plants.length}`);
    
    let inserted = 0;
    let manualCount = 0;
    
    for (const plant of plants) {
        const ecology = generateEcology(plant);
        const isManual = !!coreEcology[plant.scientific_name];
        
        await db.execute(`
            INSERT INTO plant_ecology 
            (plant_id, light_tolerance, water_requirement, temperature_tolerance, 
             drought_tolerance, cold_tolerance, data_source)
            VALUES (?, ?, ?, ?, ?, ?, ?)
            ON DUPLICATE KEY UPDATE
            light_tolerance = VALUES(light_tolerance),
            water_requirement = VALUES(water_requirement),
            temperature_tolerance = VALUES(temperature_tolerance),
            drought_tolerance = VALUES(drought_tolerance),
            cold_tolerance = VALUES(cold_tolerance),
            data_source = VALUES(data_source)
        `, [
            plant.id,
            ecology.light_tolerance,
            ecology.water_requirement,
            ecology.temperature_tolerance,
            ecology.drought_tolerance,
            ecology.cold_tolerance,
            isManual ? 'manual' : 'rule'
        ]);
        
        inserted++;
        if (isManual) manualCount++;
        
        if (inserted % 50 === 0) {
            console.log(`  已处理 ${inserted} 条`);
        }
    }
    
    console.log('\n---');
    console.log(`✅ 成功插入: ${inserted} 条`);
    console.log(`📝 手工标注: ${manualCount} 条`);
    console.log(`🤖 规则生成: ${inserted - manualCount} 条`);
    console.log(`完成时间: ${new Date().toLocaleString()}`);
}

populateEcology().catch(console.error);
